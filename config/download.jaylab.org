server {
    listen   80 ;
    server_name www.ja;

    access_log  /home/jay/log/jayproxy/access.log;

    location /download/ {
        add_header Content-Disposition "attachment";
        root /home/jay/websites/jayproxy/download/;
        rewrite ^/download/(.*)$ /$1 break;
    }

    location /yt/ {
        add_header Content-Disposition "attachment";
        root /home/jay/websites/jayproxy/;
    }

    location = /jayproxy.pac {
        root /home/jay/websites/jayproxy/pac/;
        rewrite ^/(.*)$ /$1 break;
    }
    
    location /media/ {
        root /usr/local/lib/python2.6/dist-packages/Django-1.3.1-py2.6.egg/django/contrib/admin/media/;
        rewrite ^/media/(.*)$ /$1 break;
    }

    location /site_media/ {
        root /home/jay/websites/jayproxy/site_media;
        rewrite ^/site_media/(.*)$ /$1 break;
    }

    location / {
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header Host $http_host;
        proxy_redirect off;

        proxy_pass http://127.0.0.1:8000;
    }
}
